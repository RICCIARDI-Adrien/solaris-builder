




set -e




gsed -i "s/PermitRootLogin no/PermitRootLogin yes/" /etc/ssh/sshd_config

gsed -i "s/#PermitEmptyPasswords no/PermitEmptyPasswords yes/" /etc/ssh/sshd_config


echo 'AcceptEnv   *'  >> /etc/ssh/sshd_config

mkdir -p ~/.ssh

chmod -R 600 ~/.ssh

ssh-keygen -t rsa -f ~/.ssh/id_rsa -q -N ""

echo "StrictHostKeyChecking=accept-new" >>~/.ssh/config

svcadm disable ssh

svcadm enable ssh




if [ ! -e "/etc/nsswitch.conf" ]; then
  cp /etc/nsswitch.dns  /etc/nsswitch.conf
  nscfg import name-service/switch
  svcadm refresh name-service/switch
  svcadm restart name-service/switch
fi


if [ ! -e "/etc/resolv.conf" ]; then
  echo "nameserver      8.8.8.8" >"/etc/resolv.conf"
  nscfg import dns/client
  svcadm refresh dns/client
  svcadm restart dns/client
fi



cp .bashrc .bash_profile
gsed -i 's|#SUPATH=/usr/bin:/usr/sbin|SUPATH=/usr/bin:/usr/sbin:/opt/csw/bin|' /etc/default/login
